var net = require('net'); var count = 0; //状态以及记录连接情况 //为了能偶向其他连接进来的客户端发送和广播消息，我们需要对该状态进行拓展，来追纵到底谁进来啦 //首先我们要记录设置了昵称的用户，为此要引入新的状态变量，users var users = {}; // //然后在每个连接中，在引入一个nickname的变量。 // var nickname; //createServer var server = net.createServer(function(conn){ var nickname; //handle connection console.log('\033[90m new Connections!\033[39m'); conn.write( '\n> welcome to \033[92m node-chat\033[39m!' + '\n> ' + count + ' other people are connected at this time.' + '\n> please write your name and press enter: ' ); count++; function broadcast(msg, exceptMyself){ for(var i in users){ if (!exceptMyself || i != nickname){ users[i].write(msg); } } } conn.on('data', function(data){ //console.log(data); //接收数据是，确保将回车键清除 data = data.replace('\r\n', '') if(!nickname){ if(users[data]){ conn.write('\033[93m> nickname already in use.try again: \033[39m '); return; }else{ nickname = data; users[nickname] = conn; for (var i in users){ users[i].write('\033[90m > ' + nickname + ' joined the room \033[39m \n'); } } }else{ //否则视为聊天消息，发给除自己意外的所有客户端。 for(var i in users){ if( i!= nickname){ users[i].write('\033[96m >' + nickname + ':\033]39m ' + data + '\n' ); } } } }); conn.setEncoding('utf8'); conn.on('close', function(){ count--; console.log('One Connection quit!!') delete users[nickname]; }); }); //listen server.listen(3000, function(){ console.log('\033[96m server listening on *:3000\033[39m'); });
